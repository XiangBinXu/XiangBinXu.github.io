using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp4
{
class Poker //扑克牌类
{
string[] card =
{
“红桃A”, “红桃2”, “红桃3”, “红桃4”, “红桃5”, “红桃6”, “红桃7”,
“红桃8”, “红桃9”, “红桃10”, “红桃J”, “红桃Q”, “红桃K”,
“黑桃A”, “黑桃2”, “黑桃3”, “黑桃4”, “黑桃5”, “黑桃6”, “黑桃7”,
“黑桃8”, “黑桃9”, “黑桃10”, “黑桃J”, “黑桃Q”, “黑桃K”,
“梅花A”, “梅花2”, “梅花3”, “梅花4”, “梅花5”, “梅花6”, “梅花7”,
“梅花8”, “梅花9”, “梅花10”, “梅花J”, “梅花Q”, “梅花K”,
“方片A”, “方片2”, “方片3”, “方片4”, “方片5”, “方片6”, “方片7”,
“方片8”, “方片9”, “方片10”, “方片J”, “方片Q”, “方片K”
};
public void show() //展示扑克牌
{
Console.WriteLine(“这是一幅扑克牌：”);
for (int i = 0; i < 52; i++)
{
Console.Write(card[i] + “ “);
if (i % 13 == 0)
Console.Write(“\n”); //一行13张牌
}
Console.Write(“\n”);
}
public void randomcards()//洗牌
{
Random rand = new Random();
for (int i = 0; i < 52; i++)
{
int n = rand.Next(52);
string temp;
temp = card[n];
card[n] = card[51 - n];
card[51 - n] = temp;
}
}
int position = 0; //发牌索引下标
public String deal()//发牌，按顺序发牌，从下标0开始；
{
String c = card[position];
position++;
return c;
}
}
class Player //玩家类
{
public int shownumber(String num) //计算牌的点数
{
char word = num[2];
if (word == ‘A’) //A的默认点数为11
{
return 11;
}
else if (word == ‘J’ || word == ‘Q’ || word == ‘K’ || word == ‘1’) //这里的1表示word==10
{
return 10;
}
else
{
return (int)word - 48; //数字的AScall值-48
}
}
public bool YorN(int sump) //判断玩家是否要牌
{
bool needornot = true;
string anser = Console.ReadLine();
if (anser == “yes”)
{
Console.WriteLine(“玩家继续要牌”);
needornot = true;
}
else if (anser == “not”)
{
Console.WriteLine(“玩家不要牌了”);
Console.WriteLine(“玩家的点数为：” + sump);
needornot = false;
}
return needornot;
}
}
class Computer //庄家类
{
public int shownumber(string num) //显示牌的点数
{
char word = num[2];
if (word == ‘A’) //A的默认点数为11
{
return 11;
}
else if (word == ‘J’ || word == ‘Q’ || word == ‘K’ || word == ‘1’) //这里的1表示word==10
{
return 10;
}
else
{
return (int)word - 48; //数字的AScall值-48
}
}
public bool YorN(int sumz, int sump) //判断庄家是否要牌
{
bool needornot = true;
//如果庄家小于13，直接叫牌；如果>13<=18,庄家总点数小于玩家，叫牌，总点数大于玩家，不叫牌；如果>15不叫牌
if (sumz <= 13)
{
Console.WriteLine(“庄家继续要牌”);//庄家的牌小于11，太小了，继续要牌
needornot = true;
}
else if (sumz <= 18)
{
if (sumz < sump)
{
Console.WriteLine(“庄家继续要牌”);
needornot = true;
}
else
{
Console.WriteLine(“庄家不要牌了”);
needornot = false;
}
}
else
{
Console.WriteLine(“庄家不要牌了”);
needornot = false;
}
return needornot; //返回要牌情况
}
}
public class StartGame
{
public void play()//开始游戏
{
int sump = 0; //初始化玩家的点数
int sumz = 0; //初始化庄家的点数

        string[] p = new string[8];//储存玩家的牌
        string[] z = new string[8];//储存庄家的牌



        bool pneed = true;//玩家是否要牌
        bool zneed = true;//庄家是否要牌

        bool pgo = true;//玩家能否继续要牌
        bool zgo = true;//庄家能否继续要牌

       /* bool p1judge = true;//检测玩家的牌1是否测试过
        bool p2judge = true;//检测玩家的牌2是否测试过

        bool z1judge = true;//检测庄家的牌1是否测试过
        bool z2judge = true;//检测庄家的牌2是否测试过*/

        Poker card = new Poker();//创建一幅扑克牌
        card.show(); //展示牌
        card.randomcards(); //洗牌

        Player player = new Player();//创建一个玩家
        Computer zhuangjia = new Computer();//创建一个庄家

        int i = 1;   //每局的计数器i,每局自增1               
        string str = "是";   //用来获取用户输入
        while (str == "是")
        {
            var numofpA = 0;//记录玩家A的个数
            var numofzA = 0;//记录庄家A的个数

            Console.WriteLine();  //输出空行
            Console.WriteLine("现在开始21点游戏！");
            Console.WriteLine("第" + i++.ToString() + "局:");
            Console.WriteLine("开始发牌：");
            p[0] = card.deal();//给玩家第一张牌
            z[0] = card.deal();//给庄家第一张牌

            p[1] = card.deal();//给玩家第二张牌
            z[1] = card.deal();//给庄家第二张牌

            if (player.shownumber(p[0]) == 11) { numofpA++; }
            if (player.shownumber(p[1]) == 11) { numofpA++; }
            if (zhuangjia.shownumber(z[0]) == 11) { numofzA++; }
            if (zhuangjia.shownumber(z[1]) == 11) { numofzA++; }

            sump = player.shownumber(p[0]) + player.shownumber(p[1]); //计算玩家两张牌的总数
            sumz = zhuangjia.shownumber(z[0]) + zhuangjia.shownumber(z[1]);//计算庄家两张牌的总数

            Console.WriteLine("玩家第一张牌为：" + p[0] + " 第二张牌为：" + p[1] + ",玩家手中的A个数为：" + numofpA);
            Console.WriteLine("庄家第一张牌为：" + z[0] + " 第二张牌为：" + z[1] + ",庄家手中的A个数为：" + numofzA);

            var iswing = 3;    //0代表庄家赢，1代表玩家赢，2代表平手，3代表未判读输赢

            //判断是否天胡
            if (sump == 21)   //当玩家前两张牌21点时
            {
                if (sumz == 21)  //庄家也21点就平局
                {
                    iswing = 2;
                }
                else    //否则玩家赢
                {
                    iswing = 1;
                }
            }
            if (sumz == 21)   //当庄家21点时
            {
                if (sump == 21)  //玩家也21点就平局
                {
                    iswing = 2;
                }
                else   //否则庄家赢
                {
                    iswing = 0;
                }
            }
            //处理输赢
            if (iswing == 1)
            {
                Console.WriteLine("你赢了了！21点，天胡！YOU WIN！！！");
                //TODO:让用户输入是否进行下一次循环
                Console.WriteLine("是否进行下一局？(是/否)");
                str = Console.ReadLine();
                //continue 英语继续的意思，在这里是关键字
                //代表跳出当前这次循环，执行下一次循环
                //continue之后，本次循环的语句不再执行
                continue;
            }
            else if (iswing == 0)
            {
                Console.WriteLine("庄家赢了！21点，天胡！YOU LOSE！！！");
                //TODO:让用户输入是否进行下一次循环
                Console.WriteLine("是否进行下一局？(是/否)");
                str = Console.ReadLine();
                //continue 英语继续的意思，在这里是关键字
                //代表跳出当前这次循环，执行下一次循环
                //continue之后，本次循环的语句不再执行
                continue;
            }
            else if (iswing == 2)
            {
                Console.ForegroundColor = ConsoleColor.White;
                Console.WriteLine("都是21点，天胡！平局！！！");
                //TODO:让用户输入是否进行下一次循环
                Console.WriteLine("是否进行下一局？(是/否)");
                str = Console.ReadLine();
                //continue 英语继续的意思，在这里是关键字
                //代表跳出当前这次循环，执行下一次循环
                //continue之后，本次循环的语句不再执行
                continue;
            }

            while (pgo || zgo)
            {
                //依次发牌问题
                Console.WriteLine("玩家是否要牌？（输入yes或则no）");
                pneed = player.YorN(sump);
                int j = 1;
                if (pneed && pgo)
                {
                    j++;                  
                    p[j] = card.deal();//给玩家发第j张牌
                    sump += player.shownumber(p[j]); //计算玩家的总数
                    Console.WriteLine("玩家的第" + j+1 + "张牌是：" + p[j]);
                    if (player.shownumber(p[j]) == 11) { numofpA++; }

                    pgo = true;

                    if (sump > 21)
                    {
                        if (numofpA >= 1)  //如果玩家爆点但是手里有A，那么此时拿一张A作为1
                        {
                            sump = sump - 10;
                            pgo = true;
                        }
                        else
                        {
                            pgo = false; //玩家爆点不能再取牌                               
                        }
                    }
                    else
                    {
                        pgo = true;
                    }
                }
                else
                {
                    pgo = false;  //玩家不需要取牌则玩家不能取牌了
                }

                Console.WriteLine("庄家是否要牌？（输入yes或则no）");
                zneed = zhuangjia.YorN(sumz, sump);
                int k = 1;
                if (zneed && zgo)
                {
                    k++;
                    z[k] = card.deal(); //给庄家发第k张牌
                    sumz += zhuangjia.shownumber(z[k]); //计算庄家的总数
                    Console.WriteLine("庄家的第" + k+1 + "张牌是：" + p[k]);
                    if (zhuangjia.shownumber(z[k]) == 11) { numofzA++; }

                    zgo = true;

                    if (sumz > 21)
                    {
                        if (numofzA >= 1)
                        {
                            sumz = sumz - 10;
                            zgo = true;
                        }
                        else
                        {
                            zgo = false; //庄家爆点，不能取牌                           

                        }
                    }
                    else
                    {
                        zgo = zhuangjia.YorN(sumz, sump);  //小于21点再行一步判断是否要牌
                    }
                }
                else
                {
                    zgo = false;  //庄家不需要取牌则不能取牌了
                }
            }
            Console.WriteLine("庄家底牌是：" + z[0]+" "+z[1]);
            Console.WriteLine("玩家点数是：" + sump + ",庄家点数是：" + sumz);

            if (sump >= sumz && sump <= 21 && sumz <= 21)  //玩家和庄家都没爆点时，玩家的点数大于庄家
            {
                Console.WriteLine("玩家点数大于庄家！玩家获胜！");
                Console.WriteLine("是否进行下一局？(是/否)");
                str = Console.ReadLine();
                continue;
            }
            else if (sump < sumz && sump <= 21 && sumz <= 21) //玩家和庄家都没爆点时，庄家的点数大于玩家
            {
                Console.WriteLine("庄家点数大于玩家！庄家获胜！");
                Console.WriteLine("是否进行下一局？(是/否)");
                str = Console.ReadLine();
                continue;
            }
            else if (sump > 21 && sumz <= 21) //玩家爆点庄家没爆
            {
                Console.WriteLine("玩家爆点啦！庄家胜利！");
                Console.WriteLine("是否进行下一局？(是/否)");
                str = Console.ReadLine();
                continue;
            }
            else if (sumz > 21 && sump <= 21) //庄家爆点玩家没爆
            {
                Console.WriteLine("庄家爆点啦！玩家胜利！");
                Console.WriteLine("是否进行下一局？(是/否)");
                str = Console.ReadLine();
                continue;
            }
            else if (sumz > 21 && sump > 21) //庄家和玩家都爆点
            {
                Console.WriteLine("庄家和玩家都爆点啦！平局！");
                Console.WriteLine("是否进行下一局？(是/否)");
                str = Console.ReadLine();
                continue;
            }
        }
    }
}
